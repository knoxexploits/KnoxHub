    rockStates[rockData.name] = false
    GrindTab:CreateToggle({
        Name = "Punch " .. rockData.name,
        CurrentValue = false,
        Flag = "Rock_" .. rockData.name:gsub(" ", ""),
        Callback = function(Value)
            rockStates[rockData.name] = Value
            if Value then
                local originalCFrame, originalSize, originalTransparency, originalCanCollide
                spawn(function()
                    local player = game.Players.LocalPlayer
                    local character = player.Character or player.CharacterAdded:Wait()
                    local leftHand = character:WaitForChild("LeftHand")
                    local rock = game.Workspace.machinesFolder:FindFirstChild(rockData.location):FindFirstChild("Rock")
                    if rock then
                        originalCFrame = rock.CFrame
                        originalSize = rock.Size
                        originalTransparency = rock.Transparency
                        originalCanCollide = rock.CanCollide
                        rock.Size = Vector3.new(2, 1, 1)
                        rock.Transparency = 1
                        for _, v in pairs(rock.rockGui:GetChildren()) do v.Visible = false end
                        rock.CanCollide = false
                        if rock:FindFirstChild("rockEmitter") then rock.rockEmitter:Destroy() end
                        if rock:FindFirstChild("hoopParticle") then rock.hoopParticle:Destroy() end
                        if rock:FindFirstChild("lavaParticle") then rock.lavaParticle:Destroy() end
                        while rockStates[rockData.name] do
                            rock.CFrame = leftHand.CFrame
                            wait()
                        end
                    end
                    if rock then
                        rock.CFrame = originalCFrame
                        rock.Size = originalSize
                        rock.Transparency = originalTransparency
                        rock.CanCollide = originalCanCollide
                        for _, v in pairs(rock.rockGui:GetChildren()) do v.Visible = true end
                    end
                end)
            end
        end,
    })

                            lastAttackTime = tick()
                        end
                    end
                end)

            local player = game.Players.LocalPlayer

RebirthTab:CreateToggle({

    CurrentValue = false,
    Flag = "AutoRebirth",
    Callback = function(Value)
        autoRebirthEnabled = Value
        confirmationGiven = false
        if Value then
            spawn(function()
                while autoRebirthEnabled do
                    local localPlayer = game.Players.LocalPlayer
                    local stats = localPlayer:FindFirstChild("leaderstats")
                    if not stats then break end
                    local currentRebirths = stats.Rebirths.Value
                    local strength = stats.Strength.Value
                    local durability = localPlayer.Durability.Value
                    local agility = localPlayer.Agility.Value
                    if stoppingPointEnabled and targetRebirth ~= -1 and currentRebirths >= targetRebirth then
                        autoRebirthEnabled = false
                        break
                    end
                    if (strength > 5000000 or durability > 5000000000 or agility > 5000000000) and not confirmationGiven then
                        local confirmed = false
                        createRebirthConfirmation(function() confirmed = true end)
                        while not confirmed do wait(0.1) end
                        confirmationGiven = true
                    end
                    if confirmationGiven or (strength <= 5000000000 and durability <= 5000000000 and agility <= 5000000000) then
                        local rebirthEvent = game:GetService("ReplicatedStorage").rEvents.rebirthRemote
                        if rebirthEvent then rebirthEvent:InvokeServer("rebirthRequest") end
                    end
                    wait(0.1)
                end
            end)
        end
    end,
})

local function autoKillLoop()
    local timeoutDuration = 2
    while isRunning do
        for _, player in ipairs(game.Players:GetPlayers()) do
            if player ~= game.Players.LocalPlayer and player.Character and player.Character:FindFirstChild("Head") then
                if not whitelist[player.Name] then
                    local character = player.Character
                    local humanoid = character:FindFirstChild("Humanoid")
                    if humanoid and humanoid.Health > 0 then
                        local head = character.Head
                        local leftHand = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("LeftHand")
                        local rightHand = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("RightHand")
                        if leftHand and rightHand then
                            local success = false
                            local startTime = os.clock()
                            repeat
                                firetouchinterest(head, leftHand, 0)
                                firetouchinterest(head, leftHand, 1)
                                firetouchinterest(head, rightHand, 0)
                                firetouchinterest(head, rightHand, 1)
                                success = humanoid.Health <= 0
                                task.wait(0.1)
                            until success or (os.clock() - startTime) > timeoutDuration
                        end
                    end
                end
            end
        end
        task.wait(0.1)
    end
end

    Callback = function(Value)
        if Value then
            if not isRunning then
                isRunning = true
                task.spawn(autoKillLoop)
            end
        else
            isRunning = false
        end
    end,
})

                local rightHand = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("RightHand")
                if blacklist[player.Name] and not whitelist[player.Name] then
                    if humanoid and humanoid.Health > 0 then
                        if leftHand and rightHand then
                            local success = false
                            local startTime = os.clock()
                            repeat
                                firetouchinterest(head, leftHand, 0)
                                firetouchinterest(head, leftHand, 1)
                                firetouchinterest(head, rightHand, 0)
                                firetouchinterest(head, rightHand, 1)
                                success = humanoid.Health <= 0
                                task.wait(0.1)
                            until success or (os.clock() - startTime) > timeoutDuration
                        end
                    end
                end
            end
        end
        task.wait(0.1)
    end
end

StatusTab:CreateSection("Player Statistics")

local updatingStats = true

local strengthGained = 0
local lastStrengthValue = nil
local rebirthsGained = 0
local lastRebirthsValue = nil
local killsGained = 0
local lastKillsValue = nil
local brawlsGained = 0
local lastBrawlsValue = nil
local goodKarmaGained = 0
local lastGoodKarmaValue = nil
local evilKarmaGained = 0
local lastEvilKarmaValue = nil
local KarmaGained = 0
local lastKarmaValue = nil
local durabilityGained = 0
local lastDurabilityValue = nil
local agilityGained = 0
local lastAgilityValue = nil

local function formatNumber(number)
    if number >= 1000000 then
        return string.format("%.1fM", number/1000000)
    elseif number >= 1000 then
        return string.format("%.1fK", number/1000)
    else
        return tostring(number)
    end
end

local StatLabel = StatusTab:CreateLabel("Loading stats...")

repeat task.wait() until game:IsLoaded()
if not game.Players.LocalPlayer.Character then
    game.Players.LocalPlayer.CharacterAdded:Wait()
end
repeat task.wait() until game.Players.LocalPlayer:FindFirstChild("leaderstats") and game.Players.LocalPlayer:FindFirstChild("goodKarma")

local function updateStats()
    if not updatingStats then return end
    local player = game.Players.LocalPlayer
    local currentStrength = player.leaderstats.Strength.Value
    local currentRebirths = player.leaderstats.Rebirths.Value
    local currentKills = player.leaderstats.Kills.Value
    local currentBrawls = player.leaderstats.Brawls.Value
    local currentGoodKarma = player.goodKarma.Value
    local currentEvilKarma = player.evilKarma.Value
    local currentKarma = currentGoodKarma + currentEvilKarma
    local currentDurability = player.Durability.Value
    local currentAgility = player.Agility.Value
    if lastStrengthValue == nil then lastStrengthValue = currentStrength
    elseif currentStrength > lastStrengthValue then strengthGained = strengthGained + (currentStrength - lastStrengthValue) end
    lastStrengthValue = currentStrength
    if lastRebirthsValue == nil then lastRebirthsValue = currentRebirths
    elseif currentRebirths > lastRebirthsValue then rebirthsGained = rebirthsGained + (currentRebirths - lastRebirthsValue) end
    lastRebirthsValue = currentRebirths
    if lastKillsValue == nil then lastKillsValue = currentKills
    elseif currentKills > lastKillsValue then killsGained = killsGained + (currentKills - lastKillsValue) end
    lastKillsValue = currentKills
    if lastBrawlsValue == nil then lastBrawlsValue = currentBrawls
    elseif currentBrawls > lastBrawlsValue then brawlsGained = brawlsGained + (currentBrawls - lastBrawlsValue) end
    lastBrawlsValue = currentBrawls
    if lastGoodKarmaValue == nil then lastGoodKarmaValue = currentGoodKarma
    elseif currentGoodKarma > lastGoodKarmaValue then goodKarmaGained = goodKarmaGained + (currentGoodKarma - lastGoodKarmaValue) end
    lastGoodKarmaValue = currentGoodKarma
    if lastEvilKarmaValue == nil then lastEvilKarmaValue = currentEvilKarma
    elseif currentEvilKarma > lastEvilKarmaValue then evilKarmaGained = evilKarmaGained + (currentEvilKarma - lastEvilKarmaValue) end
    lastEvilKarmaValue = currentEvilKarma
    if lastKarmaValue == nil then lastKarmaValue = currentKarma
    elseif currentKarma > lastKarmaValue then KarmaGained = KarmaGained + (currentKarma - lastKarmaValue) end
    lastKarmaValue = currentKarma
    if lastDurabilityValue == nil then lastDurabilityValue = currentDurability
    elseif currentDurability > lastDurabilityValue then durabilityGained = durabilityGained + (currentDurability - lastDurabilityValue) end
    lastDurabilityValue = currentDurability
    if lastAgilityValue == nil then lastAgilityValue = currentAgility
    elseif currentAgility > lastAgilityValue then agilityGained = agilityGained + (currentAgility - lastAgilityValue) end
    lastAgilityValue = currentAgility
    StatLabel:Set(string.format(
        "STR: %s (+%s)  |  DUR: %s (+%s)\nREB: %s (+%s)  |  KILLS: %s (+%s)\nBRAWLS: %s (+%s)  |  AGI: %s (+%s)",
        formatNumber(currentStrength), formatNumber(strengthGained),
        formatNumber(currentDurability), formatNumber(durabilityGained),
        formatNumber(currentRebirths), formatNumber(rebirthsGained),
        formatNumber(currentKills), formatNumber(killsGained),
        formatNumber(currentBrawls), formatNumber(brawlsGained),
        formatNumber(currentAgility), formatNumber(agilityGained)
    ))
end

game:GetService("RunService").RenderStepped:Connect(updateStats)

        updatingStats = Value
        if not Value then
            StatLabel:Set("Stats disabled to reduce lag.")
