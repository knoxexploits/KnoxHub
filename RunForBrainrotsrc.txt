        return

end)

-- [[ AUTO FARM LOGIC ]]
task.spawn(function()
    while true do
        task.wait(0.5)
        if autoFarmEnabled then
            local char = player.Character
            local root = char and char:FindFirstChild("HumanoidRootPart")
            if root then
                local targetPos = locations[selectedFarmArea]
                player:RequestStreamAroundAsync(targetPos)
                char:PivotTo(CFrame.new(targetPos + Vector3.new(0, 5, 0)))
                task.wait(1.5)

                local nearestPrompt = nil
                local shortestDistance = math.huge
                for _, v in pairs(workspace:GetDescendants()) do
                    if v:IsA("ProximityPrompt") then
                        local dist = (root.Position - v.Parent:GetPivot().Position).Magnitude
                        if dist < shortestDistance and dist < 150 then
                            shortestDistance = dist
                            nearestPrompt = v
                        end
                    end
                end

                if nearestPrompt then
                    char:PivotTo(nearestPrompt.Parent:GetPivot() * CFrame.new(0, 2, 3))
                    task.wait(0.3)
                    fireproximityprompt(nearestPrompt)
                    task.wait(0.5)
                end

                char:PivotTo(CFrame.new(locations["Safe Zone"] + Vector3.new(0, 5, 0)))
                
                for i = 5, 1, -1 do
                    if not autoFarmEnabled then break end
                    task.wait(1)
                end
            end
        end
    end
end)

local function updateESP()
	if not espBall or not serverPosition then return end

	espBall.CFrame = CFrame.new(serverPosition)

	colorPhase = colorPhase + 0.08
	local t = (math.sin(colorPhase) + 1) * 0.5
	espBall.Color = Color3.fromRGB(255 * t, 255 * t, 255 * t)

	orbitAngle = orbitAngle + ORBIT_SPEED
	for _, data in ipairs(orbitParticles) do
		local angle = orbitAngle + data.AngleOffset
		local x = math.cos(angle) * ORBIT_RADIUS
		local z = math.sin(angle) * ORBIT_RADIUS
		data.Part.CFrame = CFrame.new(serverPosition + Vector3.new(x, 0, z))
	end
end

local function setupTracking(h)
	if positionUpdateConnection then positionUpdateConnection:Disconnect() end
	serverPosition = h.Position
	positionUpdateConnection = h:GetPropertyChangedSignal("Position"):Connect(function()
		task.wait(0.15)
		local c = player.Character
		local rp = c and c:FindFirstChild("HumanoidRootPart")
		if rp then serverPosition = rp.Position end
	end)
end

local function initESP()
	if not espBall then createESP() end
	local c = player.Character
	local rp = c and c:FindFirstChild("HumanoidRootPart")
	if rp then setupTracking(rp) end
end

local function destroyESP()
	if espBall then
		espBall:Destroy()
		espBall = nil
	end
	for _, d in ipairs(orbitParticles) do
		d.Part:Destroy()
	end
	orbitParticles = {}
	if positionUpdateConnection then
		positionUpdateConnection:Disconnect()
		positionUpdateConnection = nil
	end
	serverPosition = nil
	orbitAngle = 0
	colorPhase = 0
end
